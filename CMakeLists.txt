cmake_minimum_required(VERSION 3.21)
project(image_codec CXX)
add_compile_definitions(_USE_MATH_DEFINES NOMINMAX )
set(CMAKE_CXX_STANDARD 17)
add_executable(crange crange.cpp range.hpp)
add_executable(drange drange.cpp range.hpp)
# Or use the header-only version
find_package(fmt CONFIG REQUIRED)
find_package(Stb REQUIRED)
target_link_libraries(crange PRIVATE fmt::fmt-header-only)
target_link_libraries(drange PRIVATE fmt::fmt-header-only)
target_include_directories(crange PRIVATE ${Stb_INCLUDE_DIR})
target_include_directories(drange PRIVATE ${Stb_INCLUDE_DIR})
